create_brennan_data <- function() {
  n_persons <- 10
  n_tasks <- 3
  n_raters_per_task <- 4
  
  var_person <- 0.37
  var_task <- 0.05
  var_rater_in_task <- 0.10
  var_person_task <- 0.15
  var_residual <- 0.40
  
  data <- expand.grid(
    Person = factor(1:n_persons),
    Task = factor(1:n_tasks),
    Rater = factor(1:n_raters_per_task))
  
  # ランダム効果の生成
  person_effects <- rnorm(n_persons, 0, sqrt(var_person))
  task_effects <- rnorm(n_tasks, 0, sqrt(var_task))
  
  # ネストされた評価者効果（タスクごとに異なる）
  rater_effects <- matrix(
    rnorm(n_tasks * n_raters_per_task, 0, sqrt(var_rater_in_task)),
    nrow = n_tasks, ncol = n_raters_per_task)
  
  # Person×Task交互作用
  person_task_effects <- matrix(
    rnorm(n_persons * n_tasks, 0, sqrt(var_person_task)),
    nrow = n_persons, ncol = n_tasks)
  
  # スコアの計算
  data$Score <- NA
  for(i in 1:nrow(data)) {
    p <- as.numeric(data$Person[i])
    t <- as.numeric(data$Task[i])
    r <- as.numeric(data$Rater[i])
    
    data$Score[i] <- 5 +  # 全体平均
      person_effects[p] +
      task_effects[t] +
      rater_effects[t, r] +
      person_task_effects[p, t] +
      rnorm(1, 0, sqrt(var_residual))
  }
  
  # 評価者IDをタスクにネストされた形式に変更
  data$Rater <- factor(paste0(data$Task, "_", data$Rater))
  
  return(data)
}

# lme4::lmer の VarCorr から G theoryで使う分散成分テーブルを作る
extract_variance_components <- function(model) {
  vc <- VarCorr(model)

  # 各分散成分（存在しない場合は0で安全に処理）
  get_v <- function(name) {
    if (name %in% names(vc)) as.numeric(vc[[name]]) else 0
  }

  var_person <- get_v("Person")
  var_task <- get_v("Task")
  var_rater <- get_v("Rater")
  var_person_task <- get_v("Person:Task")
  var_residual <- attr(vc, "sc")^2

  total <- var_person + var_task + var_rater + var_person_task + var_residual

  # グルーピング因子の水準数（n）も併記
  fl <- lme4::getME(model, "flist")
  n_person <- if ("Person" %in% names(fl)) nlevels(fl[["Person"]]) else NA_integer_
  n_task <- if ("Task" %in% names(fl)) nlevels(fl[["Task"]]) else NA_integer_
  n_rater <- if ("Rater" %in% names(fl)) nlevels(fl[["Rater"]]) else NA_integer_
  n_person_task <- if ("Person:Task" %in% names(fl)) nlevels(fl[["Person:Task"]]) else NA_integer_

  tibble::tibble(
    source  = factor(
      c("Person", "Task", "Rater", "Person:Task", "Residual"),
      levels = c("Person", "Task", "Rater", "Person:Task", "Residual")),
    var     = c(var_person, var_task, var_rater, var_person_task, var_residual),
    percent = 100 * var / total,
    n       = c(n_person, n_task, n_rater, n_person_task, NA_integer_))
}

15.1 信頼性係数の計算関数

# G係数（相対的決定）の計算
calculate_g_coefficient <- function(var_components, n_tasks, n_raters) {
  var_p <- var_components$var[var_components$source == "Person"]
  var_t <- var_components$var[var_components$source == "Task"]
  var_r <- var_components$var[var_components$source == "Rater"]
  var_pt <- var_components$var[var_components$source == "Person:Task"]
  var_e <- var_components$var[var_components$source == "Residual"]
  
  # 相対的誤差分散（交互作用のみ）
  var_rel <- var_pt / n_tasks + var_e / (n_tasks * n_raters)
  
  # G係数
  g_coef <- var_p / (var_p + var_rel)
  
  return(g_coef)
}

# φ係数（絶対的決定）の計算
calculate_phi_coefficient <- function(var_components, n_tasks, n_raters) {
  var_p <- var_components$var[var_components$source == "Person"]
  var_t <- var_components$var[var_components$source == "Task"]
  var_r <- var_components$var[var_components$source == "Rater"]
  var_pt <- var_components$var[var_components$source == "Person:Task"]
  var_e <- var_components$var[var_components$source == "Residual"]
  
  # 絶対的誤差分散（すべての誤差源）
  var_abs <- var_t / n_tasks + var_r / (n_tasks * n_raters) + 
             var_pt / n_tasks + var_e / (n_tasks * n_raters)
  
  # φ係数
  phi_coef <- var_p / (var_p + var_abs)
  
  return(phi_coef)
}

# シミュレーション実行
d_study_simulation <- function(var_components, max_tasks = 5, max_raters = 10) {
  results <- tibble()
  
  for (n_tasks in 1:max_tasks) {
    for (n_raters in 1:max_raters) {
      g_coef <- calculate_g_coefficient(var_components, n_tasks, n_raters)
      phi_coef <- calculate_phi_coefficient(var_components, n_tasks, n_raters)
      
      results <- bind_rows(
        results,
        tibble(
          TaskCount = n_tasks,
          RaterCount = n_raters,
          Generalizability = g_coef,
          Dependability = phi_coef
        )
      )
    }
  }
  
  results <- results |>
    mutate(
      TaskCount = factor(TaskCount),
      RaterCount = factor(RaterCount)
    )
  
  return(results)
}

# Rajaratnam.2相当のデータを生成
create_rajaratnam_data <- function() {
  n_persons <- 20
  n_items_per_subtest <- c(5, 4, 6)  # サブテストごとの項目数
  n_subtests <- 3
  
  # 分散成分の設定（サブテストごと）
  var_person <- c(0.35, 0.40, 0.38)
  var_item <- c(0.08, 0.10, 0.09)
  var_residual <- c(0.30, 0.35, 0.32)
  
  # 相関行列（Person間の相関）
  person_cor <- matrix(c(
    1.00, 0.65, 0.60,
    0.65, 1.00, 0.70,
    0.60, 0.70, 1.00
  ), nrow = 3)
  
  data_list <- list()
  
  for (s in 1:n_subtests) {
    # データグリッド
    subtest_data <- expand.grid(
      Person = factor(1:n_persons),
      Item = factor(1:n_items_per_subtest[s]),
      Subtest = factor(s)
    )
    
    # ランダム効果の生成
    person_effects <- rnorm(n_persons, 0, sqrt(var_person[s]))
    item_effects <- rnorm(n_items_per_subtest[s], 0, sqrt(var_item[s]))
    
    # スコアの計算
    subtest_data$Score <- NA
    for(i in 1:nrow(subtest_data)) {
      p <- as.numeric(subtest_data$Person[i])
      it <- as.numeric(subtest_data$Item[i])
      
      subtest_data$Score[i] <- 5 +
        person_effects[p] +
        item_effects[it] +
        rnorm(1, 0, sqrt(var_residual[s]))
    }
    
    data_list[[s]] <- subtest_data
  }
  
  # データの結合
  Rajaratnam.2 <- bind_rows(data_list)
  
  return(Rajaratnam.2)
}

# サブテストごとにlme4モデルを適用
multivariate_g_study <- function(data) {
  subtests <- unique(data$Subtest)
  results <- list()
  
  for (s in subtests) {
    subtest_data <- data |> filter(Subtest == s)
    
    # lme4モデル
    model <- lmer(
      Score ~ 1 + (1|Person) + (1|Item),
      data = subtest_data,
      REML = TRUE
    )
    
    # 分散成分の抽出
    vc <- VarCorr(model)
    var_person <- as.numeric(vc$Person)
    var_item <- as.numeric(vc$Item)
    var_residual <- attr(vc, "sc")^2
    var_total <- var_person + var_item + var_residual
    
    results[[paste0("Subtest", s)]] <- data.frame(
      Subtest = s,
      source = c("Person", "Item", "Residual"),
      var = c(var_person, var_item, var_residual),
      percent = 100 * c(var_person, var_item, var_residual) / var_total
    )
  }
  
  return(results)
}

estimate_subtest_covariance <- function(data, mv_results) {
  subtests <- unique(data$Subtest)
  n_subtests <- length(subtests)
  
  # 各サブテストのPerson分散を取得
  person_vars <- numeric(n_subtests)
  for (i in 1:n_subtests) {
    subtest_result <- mv_results[[paste0("Subtest", i)]]
    person_vars[i] <- subtest_result$var[subtest_result$source == "Person"]
  }
  
  # Person効果の推定値を取得
  person_effects <- matrix(NA, nrow = length(unique(data$Person)), ncol = n_subtests)
  
  for (s in 1:n_subtests) {
    subtest_data <- data |> filter(Subtest == s)
    model <- lmer(Score ~ 1 + (1|Person) + (1|Item), data = subtest_data, REML = TRUE)
    person_effects[, s] <- ranef(model)$Person[[1]]
  }
  
  # 共分散行列の計算
  cov_matrix <- cov(person_effects)
  colnames(cov_matrix) <- paste0("Subtest", 1:n_subtests)
  rownames(cov_matrix) <- paste0("Subtest", 1:n_subtests)
  
  return(list(
    covariance = cov_matrix,
    correlation = cov2cor(cov_matrix),
    person_variances = person_vars
  ))
}

# 多変量信頼性係数の計算
calculate_composite_reliability <- function(mv_results, cov_matrix, weights, n_items) {
  n_subtests <- length(weights)
  
  # 各サブテストの分散成分を取得
  person_vars <- numeric(n_subtests)
  item_vars <- numeric(n_subtests)
  residual_vars <- numeric(n_subtests)
  
  for (i in 1:n_subtests) {
    subtest_result <- mv_results[[paste0("Subtest", i)]]
    person_vars[i] <- subtest_result$var[subtest_result$source == "Person"]
    item_vars[i] <- subtest_result$var[subtest_result$source == "Item"]
    residual_vars[i] <- subtest_result$var[subtest_result$source == "Residual"]
  }
  
  # 合成スコアの分散（Universe分散）
  var_universe <- 0
  for (i in 1:n_subtests) {
    for (j in 1:n_subtests) {
      if (i == j) {
        var_universe <- var_universe + weights[i]^2 * person_vars[i]
      } else {
        var_universe <- var_universe + weights[i] * weights[j] * cov_matrix[i, j]
      }
    }
  }
  
  # 相対的誤差分散
  var_rel <- 0
  for (i in 1:n_subtests) {
    var_rel <- var_rel + weights[i]^2 * residual_vars[i] / n_items[i]
  }
  
  # 絶対的誤差分散
  var_abs <- var_rel
  for (i in 1:n_subtests) {
    var_abs <- var_abs + weights[i]^2 * item_vars[i] / n_items[i]
  }
  
  # 信頼性係数
  generalizability <- var_universe / (var_universe + var_rel)
  dependability <- var_universe / (var_universe + var_abs)
  
  # SEM
  sem_rel <- sqrt(var_rel)
  sem_abs <- sqrt(var_abs)
  
  return(list(
    var_universe = var_universe,
    generalizability = generalizability,
    dependability = dependability,
    sem_rel = sem_rel,
    sem_abs = sem_abs
  ))
}

simulate_mv_reliability <- function(mv_results, cov_matrix, weights, max_items = 10) {
  n_subtests <- length(weights)
  results <- tibble()
  
  # 基本の分散成分を取得
  person_vars <- numeric(n_subtests)
  item_vars <- numeric(n_subtests)
  residual_vars <- numeric(n_subtests)
  
  for (i in 1:n_subtests) {
    subtest_result <- mv_results[[paste0("Subtest", i)]]
    person_vars[i] <- subtest_result$var[subtest_result$source == "Person"]
    item_vars[i] <- subtest_result$var[subtest_result$source == "Item"]
    residual_vars[i] <- subtest_result$var[subtest_result$source == "Residual"]
  }
  
  # 項目数を変えてシミュレーション
  for (n_items_base in 1:max_items) {
    # 各サブテストの項目数（比率を保つ）
    n_items <- c(n_items_base, 
                 round(n_items_base * 0.8), 
                 round(n_items_base * 1.2))
    n_items[n_items == 0] <- 1  # 最低1項目
    
    # Universe分散
    var_universe <- 0
    for (i in 1:n_subtests) {
      for (j in 1:n_subtests) {
        if (i == j) {
          var_universe <- var_universe + weights[i]^2 * person_vars[i]
        } else {
          var_universe <- var_universe + weights[i] * weights[j] * cov_matrix[i, j]
        }
      }
    }
    
    # 誤差分散
    var_rel <- 0
    var_abs <- 0
    for (i in 1:n_subtests) {
      var_rel <- var_rel + weights[i]^2 * residual_vars[i] / n_items[i]
      var_abs <- var_abs + weights[i]^2 * (item_vars[i] + residual_vars[i]) / n_items[i]
    }
    
    # 信頼性係数
    g_coef <- var_universe / (var_universe + var_rel)
    phi_coef <- var_universe / (var_universe + var_abs)
    
    results <- bind_rows(
      results,
      tibble(
        ItemCount = n_items_base,
        Items_S1 = n_items[1],
        Items_S2 = n_items[2],
        Items_S3 = n_items[3],
        Generalizability = g_coef,
        Dependability = phi_coef
      )
    )
  }
  
  return(results)
}

# ICC (Intraclass Correlation Coefficient) の計算
calculate_icc <- function(model) {
  vc <- VarCorr(model)
  
  # 各レベルのICC
  var_person <- as.numeric(vc$Person)
  var_task <- as.numeric(vc$Task)
  var_rater <- as.numeric(vc$Rater)
  var_person_task <- as.numeric(vc$`Person:Task`)
  var_residual <- attr(vc, "sc")^2
  var_total <- var_person + var_task + var_rater + var_person_task + var_residual
  
  icc_results <- data.frame(
    Level = c("Person", "Task", "Rater", "Person:Task"),
    ICC = c(
      var_person / var_total,
      var_task / var_total,
      var_rater / var_total,
      var_person_task / var_total
    )
  )
  
  return(icc_results)
}

#| label: create-brennan-data

create_brennan_data <- function() {
  n_persons <- 10
  n_tasks <- 3
  n_raters_per_task <- 4
  
  var_person <- 0.37
  var_task <- 0.05
  var_rater_in_task <- 0.10
  var_person_task <- 0.15
  var_residual <- 0.40
  
  data <- expand.grid(
    Person = factor(1:n_persons),
    Task = factor(1:n_tasks),
    Rater = factor(1:n_raters_per_task))
  
  # ランダム効果の生成
  person_effects <- rnorm(n_persons, 0, sqrt(var_person))
  task_effects <- rnorm(n_tasks, 0, sqrt(var_task))
  
  # ネストされた評価者効果（タスクごとに異なる）
  rater_effects <- matrix(
    rnorm(n_tasks * n_raters_per_task, 0, sqrt(var_rater_in_task)),
    nrow = n_tasks, ncol = n_raters_per_task)
  
  # Person×Task交互作用
  person_task_effects <- matrix(
    rnorm(n_persons * n_tasks, 0, sqrt(var_person_task)),
    nrow = n_persons, ncol = n_tasks)
  
  # スコアの計算
  data$Score <- NA
  for(i in 1:nrow(data)) {
    p <- as.numeric(data$Person[i])
    t <- as.numeric(data$Task[i])
    r <- as.numeric(data$Rater[i])
    
    data$Score[i] <- 5 +  # 全体平均
      person_effects[p] +
      task_effects[t] +
      rater_effects[t, r] +
      person_task_effects[p, t] +
      rnorm(1, 0, sqrt(var_residual))
  }
  
  # 評価者IDをタスクにネストされた形式に変更
  data$Rater <- factor(paste0(data$Task, "_", data$Rater))
  
  return(data)
}

# lme4::lmer の VarCorr から G theoryで使う分散成分テーブルを作る
extract_variance_components <- function(model) {
  vc <- VarCorr(model)

  # 各分散成分（存在しない場合は0で安全に処理）
  get_v <- function(name) {
    if (name %in% names(vc)) as.numeric(vc[[name]]) else 0
  }

  var_person <- get_v("Person")
  var_task <- get_v("Task")
  var_rater <- get_v("Rater")
  var_person_task <- get_v("Person:Task")
  var_residual <- attr(vc, "sc")^2

  total <- var_person + var_task + var_rater + var_person_task + var_residual

  # グルーピング因子の水準数（n）も併記
  fl <- lme4::getME(model, "flist")
  n_person <- if ("Person" %in% names(fl)) nlevels(fl[["Person"]]) else NA_integer_
  n_task <- if ("Task" %in% names(fl)) nlevels(fl[["Task"]]) else NA_integer_
  n_rater <- if ("Rater" %in% names(fl)) nlevels(fl[["Rater"]]) else NA_integer_
  n_person_task <- if ("Person:Task" %in% names(fl)) nlevels(fl[["Person:Task"]]) else NA_integer_

  tibble::tibble(
    source  = factor(
      c("Person", "Task", "Rater", "Person:Task", "Residual"),
      levels = c("Person", "Task", "Rater", "Person:Task", "Residual")),
    var     = c(var_person, var_task, var_rater, var_person_task, var_residual),
    percent = 100 * var / total,
    n       = c(n_person, n_task, n_rater, n_person_task, NA_integer_))
}

## 信頼性係数の計算関数

```{r}
#| label: reliability-functions

# G係数（相対的決定）の計算
calculate_g_coefficient <- function(var_components, n_tasks, n_raters) {
  var_p <- var_components$var[var_components$source == "Person"]
  var_t <- var_components$var[var_components$source == "Task"]
  var_r <- var_components$var[var_components$source == "Rater"]
  var_pt <- var_components$var[var_components$source == "Person:Task"]
  var_e <- var_components$var[var_components$source == "Residual"]
  
  # 相対的誤差分散（交互作用のみ）
  var_rel <- var_pt / n_tasks + var_e / (n_tasks * n_raters)
  
  # G係数
  g_coef <- var_p / (var_p + var_rel)
  
  return(g_coef)
}

# φ係数（絶対的決定）の計算
calculate_phi_coefficient <- function(var_components, n_tasks, n_raters) {
  var_p <- var_components$var[var_components$source == "Person"]
  var_t <- var_components$var[var_components$source == "Task"]
  var_r <- var_components$var[var_components$source == "Rater"]
  var_pt <- var_components$var[var_components$source == "Person:Task"]
  var_e <- var_components$var[var_components$source == "Residual"]
  
  # 絶対的誤差分散（すべての誤差源）
  var_abs <- var_t / n_tasks + var_r / (n_tasks * n_raters) + 
             var_pt / n_tasks + var_e / (n_tasks * n_raters)
  
  # φ係数
  phi_coef <- var_p / (var_p + var_abs)
  
  return(phi_coef)
}
```

# シミュレーション実行
d_study_simulation <- function(var_components, max_tasks = 5, max_raters = 10) {
  results <- tibble()
  
  for (n_tasks in 1:max_tasks) {
    for (n_raters in 1:max_raters) {
      g_coef <- calculate_g_coefficient(var_components, n_tasks, n_raters)
      phi_coef <- calculate_phi_coefficient(var_components, n_tasks, n_raters)
      
      results <- bind_rows(
        results,
        tibble(
          TaskCount = n_tasks,
          RaterCount = n_raters,
          Generalizability = g_coef,
          Dependability = phi_coef
        )
      )
    }
  }
  
  results <- results |>
    mutate(
      TaskCount = factor(TaskCount),
      RaterCount = factor(RaterCount)
    )
  
  return(results)
}

multivariate_g_study <- function(data) {
  subtests <- unique(data$Subtest)
  results <- list()
  
  for (s in subtests) {
    subtest_data <- data |> filter(Subtest == s)
    
    # lme4モデル
    model <- lmer(
      Score ~ 1 + (1|Person) + (1|Item),
      data = subtest_data,
      REML = TRUE
    )
    
    # 分散成分の抽出
    vc <- VarCorr(model)
    var_person <- as.numeric(vc$Person)
    var_item <- as.numeric(vc$Item)
    var_residual <- attr(vc, "sc")^2
    var_total <- var_person + var_item + var_residual
    
    results[[paste0("Subtest", s)]] <- data.frame(
      Subtest = s,
      source = c("Person", "Item", "Residual"),
      var = c(var_person, var_item, var_residual),
      percent = 100 * c(var_person, var_item, var_residual) / var_total
    )
  }
  
  return(results)
}

estimate_subtest_covariance <- function(data, mv_results) {
  subtests <- unique(data$Subtest)
  n_subtests <- length(subtests)
  
  # 各サブテストのPerson分散を取得
  person_vars <- numeric(n_subtests)
  for (i in 1:n_subtests) {
    subtest_result <- mv_results[[paste0("Subtest", i)]]
    person_vars[i] <- subtest_result$var[subtest_result$source == "Person"]
  }
  
  # Person効果の推定値を取得
  person_effects <- matrix(NA, nrow = length(unique(data$Person)), ncol = n_subtests)
  
  for (s in 1:n_subtests) {
    subtest_data <- data |> filter(Subtest == s)
    model <- lmer(Score ~ 1 + (1|Person) + (1|Item), data = subtest_data, REML = TRUE)
    person_effects[, s] <- ranef(model)$Person[[1]]
  }
  
  # 共分散行列の計算
  cov_matrix <- cov(person_effects)
  colnames(cov_matrix) <- paste0("Subtest", 1:n_subtests)
  rownames(cov_matrix) <- paste0("Subtest", 1:n_subtests)
  
  return(list(
    covariance = cov_matrix,
    correlation = cov2cor(cov_matrix),
    person_variances = person_vars
  ))
}

calculate_composite_reliability <- function(mv_results, cov_matrix, weights, n_items) {
  n_subtests <- length(weights)
  
  # 各サブテストの分散成分を取得
  person_vars <- numeric(n_subtests)
  item_vars <- numeric(n_subtests)
  residual_vars <- numeric(n_subtests)
  
  for (i in 1:n_subtests) {
    subtest_result <- mv_results[[paste0("Subtest", i)]]
    person_vars[i] <- subtest_result$var[subtest_result$source == "Person"]
    item_vars[i] <- subtest_result$var[subtest_result$source == "Item"]
    residual_vars[i] <- subtest_result$var[subtest_result$source == "Residual"]
  }
  
  # 合成スコアの分散（Universe分散）
  var_universe <- 0
  for (i in 1:n_subtests) {
    for (j in 1:n_subtests) {
      if (i == j) {
        var_universe <- var_universe + weights[i]^2 * person_vars[i]
      } else {
        var_universe <- var_universe + weights[i] * weights[j] * cov_matrix[i, j]
      }
    }
  }
  
  # 相対的誤差分散
  var_rel <- 0
  for (i in 1:n_subtests) {
    var_rel <- var_rel + weights[i]^2 * residual_vars[i] / n_items[i]
  }
  
  # 絶対的誤差分散
  var_abs <- var_rel
  for (i in 1:n_subtests) {
    var_abs <- var_abs + weights[i]^2 * item_vars[i] / n_items[i]
  }
  
  # 信頼性係数
  generalizability <- var_universe / (var_universe + var_rel)
  dependability <- var_universe / (var_universe + var_abs)
  
  # SEM
  sem_rel <- sqrt(var_rel)
  sem_abs <- sqrt(var_abs)
  
  return(list(
    var_universe = var_universe,
    generalizability = generalizability,
    dependability = dependability,
    sem_rel = sem_rel,
    sem_abs = sem_abs
  ))
}

calculate_icc <- function(model) {
  vc <- VarCorr(model)
  
  # 各レベルのICC
  var_person <- as.numeric(vc$Person)
  var_task <- as.numeric(vc$Task)
  var_rater <- as.numeric(vc$Rater)
  var_person_task <- as.numeric(vc$`Person:Task`)

    ICC = c(
      var_person / var_total,
      var_task / var_total,
      var_rater / var_total,
      var_person_task / var_total
    )
  )
  
  return(icc_results)
}
